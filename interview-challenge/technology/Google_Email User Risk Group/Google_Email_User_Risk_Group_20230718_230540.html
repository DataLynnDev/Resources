# code block 6
from datetime import datetime

CURRENT_DATE = datetime.today()

# Start with creating a copy of the gmail_user_df DataFrame to preserve original data
gmail_user_data_enhanced = gmail_user_df.copy()

# Create a new column 'Trend_Impact_Score' initialized to 0
gmail_user_data_enhanced['Trend_Impact_Score'] = 0

# Define the current date
CURRENT_DATE = datetime.today()

# Create a mapping from impact description to numerical score
trend_impact_mapping = {'Low': 1, 'Medium': 2, 'High': 3}
external_tech_trends_df['Trend_Impact'] = external_tech_trends_df['Trend_Impact'].map(trend_impact_mapping)

# Loop over all trends
for _, trend in external_tech_trends_df.iterrows():
    # Let's create a DataFrame to store trend impact scores for users
    trend_impact_scores = pd.DataFrame(index=gmail_user_df.index, columns=['Trend_Impact_Score'])
    trend_impact_scores = trend_impact_scores.fillna(0)

    # Check if the user was active during the trend period
    account_creation_date = gmail_user_df['Account_Age'].apply(lambda x: CURRENT_DATE - pd.DateOffset(days=x))
    trend_start_date = pd.to_datetime(trend['Trend_Start_Date'])
    trend_end_date = pd.to_datetime(trend['Trend_End_Date'])

    # Convert trend_end_date to pandas Series with datetime64[ns] type
    trend_end_date = pd.Series(pd.to_datetime([trend_end_date]*len(account_creation_date)))

    active_during_trend = (account_creation_date <= trend_end_date) & (trend_start_date <= CURRENT_DATE)
    trend_impact_scores.loc[active_during_trend, 'Trend_Impact_Score'] += trend['Trend_Impact']

    # Normalize the scores
    trend_impact_scores['Trend_Impact_Score'] = (trend_impact_scores['Trend_Impact_Score'] - trend_impact_scores['Trend_Impact_Score'].min()) / (trend_impact_scores['Trend_Impact_Score'].max() - trend_impact_scores['Trend_Impact_Score'].min())

    # Add the trend impact scores to the enhanced user data DataFrame using the DataFrame.update() method
    gmail_user_data_enhanced.update(trend_impact_scores)

# Normalize the overall 'Trend_Impact_Score' in gmail_user_data_enhanced
gmail_user_data_enhanced['Trend_Impact_Score'] = (gmail_user_data_enhanced['Trend_Impact_Score'] - gmail_user_data_enhanced['Trend_Impact_Score'].min()) / (gmail_user_data_enhanced['Trend_Impact_Score'].max() - gmail_user_data_enhanced['Trend_Impact_Score'].min())

import pandas as pd

# define the function to map 'Risk_Group' to new values
def adjust_risk(row):
    if row['Trend_Impact_Score'] == 1:
        if row['Risk_Group'] == 'Low':
            return 'Medium'
        elif row['Risk_Group'] == 'Medium':
            return 'High'
        else: # 'High' stays 'High'
            return 'High'
    else:
        return row['Risk_Group']

# apply the function to the DataFrame
