import sqlite3
import pandas as pd

# For demonstration purposes, creating a dummy 'User_Details' DataFrame
user_details = pd.DataFrame({
    'user_id': list(range(1, 101)),  # Assuming there are 100 users with IDs from 1 to 100
    'user_name': ['user_' + str(i) for i in range(1, 101)]
})

# Establishing connection
conn = sqlite3.connect(':memory:')
c = conn.cursor()

# Convert DataFrames to SQL tables
df_info.to_sql('df_info', conn, if_exists='replace', index=False)
user_details.to_sql('User_Details', conn, if_exists='replace', index=False)

# 1. Rank users based on their total costs
query1 = """
    SELECT user_id, SUM(cost) as total_cost,
    RANK() OVER(ORDER BY SUM(cost) DESC) as rank
    FROM df_info
    GROUP BY user_id
    ORDER BY total_cost DESC;
"""
ranked_users = pd.read_sql_query(query1, conn)
print(ranked_users)

# 2. List of users who used the service in both 'region A' and 'region B'
query2 = """
    SELECT DISTINCT user_id
    FROM df_info
    WHERE user_id IN (SELECT user_id FROM df_info WHERE region = 'North')
    AND user_id IN (SELECT user_id FROM df_info WHERE region = 'South');
"""
users_in_both_regions = pd.read_sql_query(query2, conn)
print(users_in_both_regions)

# 3. Left and right joins with 'User_Details'
# Left Join
query3a = """
    SELECT ud.user_id, ud.user_name, id.service_id, id.usage_date
    FROM User_Details ud
    LEFT JOIN df_info id ON ud.user_id = id.user_id;
"""
left_join = pd.read_sql_query(query3a, conn)
print(left_join)

# Right Join (SQLite doesn't support RIGHT JOIN, so we'll simulate it using LEFT JOIN by switching table positions)
query3b = """
    SELECT id.user_id, ud.user_name, id.service_id, id.usage_date
    FROM df_info id
    LEFT JOIN User_Details ud ON id.user_id = ud.user_id;
"""
right_join = pd.read_sql_query(query3b, conn)
print(right_join)

# 4. CTE to fetch top 10 users with the highest performance score
query4 = """
    WITH RankedUsers AS (
        SELECT user_id, AVG(performance_score) as avg_score,
        RANK() OVER(ORDER BY AVG(performance_score) DESC) as rank
        FROM df_info
        GROUP BY user_id
    )
    SELECT user_id, avg_score
    FROM RankedUsers
    WHERE rank <= 10;
"""
top_users = pd.read_sql_query(query4, conn)
print(top_users)

# Closing the connection
conn.close()

  • Calculate the probability that a user receives more than three updates within a week.
  • Then, implement this calculation programmatically to retrieve the result.

from scipy.stats import norm

# Assuming updates are rolled out evenly across a month (let's assume 20 updates in a month for this example)
mean_updates_per_week = 20 / 4.33
# Assuming a standard deviation (just a guess here, it would typically be provided or calculated from actual data)
std_dev = 2

# Z-score for receiving more than 3 updates in a week
z = (3 - mean_updates_per_week) / std_dev

# Calculating the probability
probability = 1 - norm.cdf(z)

print(probability)
