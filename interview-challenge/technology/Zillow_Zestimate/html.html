Question 1¶

Scenario: You've been informed that some of the homes in our database may have outdated Zestimates. By analyzing the gap between LastSoldPrice and Zestimate, and considering homes sold in the last 6 months, identify homes that might have outdated Zestimates and which states have the highest percentage of such homes.

import pandas as pd
from datetime import datetime, timedelta

df = df_HomeData.copy()
# Step 1: Filter for Homes Sold in Last 6 Months
six_months_ago = datetime.now() - timedelta(days=180)
df['LastSoldDate'] = pd.to_datetime(df['LastSoldDate'])
recent_sales = df[df['LastSoldDate'] > six_months_ago].copy()  # Create a copy

# Step 2: Calculate Gap between LastSoldPrice and Zestimate
recent_sales['Gap'] = abs(recent_sales['LastSoldPrice'] - recent_sales['Zestimate'])
recent_sales['RelativeGap'] = (recent_sales['Gap'] / recent_sales['LastSoldPrice']) * 100

# Step 3: Identify Outdated Zestimates
# Setting a threshold of 10% relative gap to identify outdated Zestimates
outdated_homes = recent_sales[recent_sales['RelativeGap'] > 10]

# Step 4: Analyze by State
# Count total homes and outdated homes by state
total_homes_by_state = recent_sales['State'].value_counts()
outdated_homes_by_state = outdated_homes['State'].value_counts()

# Calculate the percentage of outdated homes by state
outdated_percentage_by_state = (outdated_homes_by_state / total_homes_by_state) * 100
outdated_percentage_by_state = outdated_percentage_by_state.sort_values(ascending=False)

# Print the homes with outdated Zestimates
print("Homes with Outdated Zestimates(Head):")
print(outdated_homes.head())

# Print the states with the highest percentage of outdated Zestimates
print("State with Highest Percentage of Outdated Zestimates:")
print(outdated_percentage_by_state.head())

import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler, OneHotEncoder
from sklearn.compose import ColumnTransformer
from sklearn.pipeline import Pipeline
from sklearn.ensemble import RandomForestRegressor
from sklearn.metrics import mean_squared_error, mean_absolute_error
import numpy as np

# Assuming HomeData is a DataFrame containing the information
# HomeData = pd.read_csv('HomeData.csv')

# Step 1: Feature Selection
features = ['PhysicalAttrib_Rooms','PhysicalAttrib_Sqft' , 'TaxAssessment', 'UrbanOrRural', 'NearbyAvgPrice', 'State', 'SavedCount']
target = ['Zestimate']

# Step 2: Data Preprocessing
# Handling Missing Data: Drop or impute missing values if needed
# HomeData.dropna(inplace=True)  # Example to drop missing values

# Encode categorical variables and normalize numerical ones
numeric_features = ['PhysicalAttrib_Rooms','PhysicalAttrib_Sqft', 'TaxAssessment', 'NearbyAvgPrice', 'SavedCount']
categorical_features = ['UrbanOrRural', 'State']

numeric_transformer = StandardScaler()
categorical_transformer = OneHotEncoder(handle_unknown='ignore')

preprocessor = ColumnTransformer(
    transformers=[
        ('num', numeric_transformer, numeric_features),
        ('cat', categorical_transformer, categorical_features)])

# Step 3: Model Selection
model = RandomForestRegressor(n_estimators=100, random_state=0)

# Combining preprocessor and the model into a single pipeline
pipeline = Pipeline(steps=[('preprocessor', preprocessor),
                           ('model', model)
                           ])

# Step 4: Train-Test Split
X = df[features]
y = df[target]
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2, random_state=0)

# Step 5: Model Training
pipeline.fit(X_train, y_train.values.ravel())

# Step 6: Model Evaluation
y_pred = pipeline.predict(X_test)
rmse = np.sqrt(mean_squared_error(y_test, y_pred))
mae = mean_absolute_error(y_test, y_pred)

print(f"Root Mean Square Error: {rmse}")
print(f"Mean Absolute Error: {mae}")
