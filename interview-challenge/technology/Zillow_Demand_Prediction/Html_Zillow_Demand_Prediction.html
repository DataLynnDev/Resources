from scipy import stats

# 1. Calculating the Z-score
info_data['Z_Score'] = np.abs(stats.zscore(info_data['Historical_Sale']))

# 2. Identifying outliers (e.g., Z-Score > 3)
outliers = info_data[info_data['Z_Score'] > 3]

# 3. Visualizing outliers
sns.scatterplot(data=outliers, x='Economic_Indicator', y='Historical_Sale')
plt.show()

from sklearn.model_selection import train_test_split, cross_val_score
from sklearn.linear_model import LinearRegression
from sklearn.ensemble import RandomForestRegressor
from sklearn.metrics import mean_squared_error
from sklearn.preprocessing import StandardScaler

# 1. Drop rows with NaN values
info_data.dropna(subset=['Sale_Date', 'Listing_Date', 'Economic_Indicator', 'Historical_Sale', 'Listing_Price', 'Demand_Score'], inplace=True)

# 2. Transforming date features
info_data['Sale_Date'] = pd.to_datetime(info_data['Sale_Date'])
info_data['Listing_Date'] = pd.to_datetime(info_data['Listing_Date'])

info_data['Sale_Year'] = info_data['Sale_Date'].dt.year
info_data['Sale_Month'] = info_data['Sale_Date'].dt.month
info_data['Sale_Day'] = info_data['Sale_Date'].dt.day

info_data['Listing_Year'] = info_data['Listing_Date'].dt.year
info_data['Listing_Month'] = info_data['Listing_Date'].dt.month
info_data['Listing_Day'] = info_data['Listing_Date'].dt.day

# 3. Feature selection and scaling
features = ['Economic_Indicator', 'Historical_Sale', 'Listing_Price', 'Sale_Year', 'Sale_Month', 'Sale_Day', 'Listing_Year', 'Listing_Month', 'Listing_Day']
X = info_data[features]
y = info_data['Demand_Score']

scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

# 4. Data Splitting and Model Training
X_train, X_test, y_train, y_test = train_test_split(X_scaled, y, test_size=0.2, random_state=42)

model = LinearRegression()
cross_val_score(model, X_train, y_train, cv=5)
model.fit(X_train, y_train)

# Validation
y_pred = model.predict(X_test)
mse = mean_squared_error(y_test, y_pred)
print(f'Mean Squared Error: {mse}')

Solution:¶

  2. Use a t-test to compare the means.

# 1. Calculating means for each Region_Type
mean_scores = info_data.groupby('Region_Type')['Demand_Score'].mean()
print(mean_scores)

# 2. T-Test
rural_scores = info_data[info_data['Region_Type'] == 'Rural']['Demand_Score']
urban_scores = info_data[info_data['Region_Type'] == 'Urban']['Demand_Score']
t_stat, p_value = stats.ttest_ind(rural_scores, urban_scores)

# 3. Interpretation
if p_value < 0.05:
    print("The means are statistically different.")
else:
    print("The means are not statistically different.")
