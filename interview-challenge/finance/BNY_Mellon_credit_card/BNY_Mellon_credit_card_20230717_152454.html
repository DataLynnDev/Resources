from sklearn.preprocessing import LabelEncoder
from sklearn.compose import make_column_transformer
from sklearn.preprocessing import OneHotEncoder
from imblearn.over_sampling import SMOTE
import numpy as np

# Create a copy to not modify the original data
customers_copy = customers.copy()

# Convert 'Yes' and 'No' in 'default_history' to numerical values
encoder = LabelEncoder()
customers_copy['default_history'] = encoder.fit_transform(customers_copy['default_history'])

# Convert datetime to Unix timestamp (number of seconds since 1970-01-01)
for col in customers_copy.select_dtypes(include=[np.datetime64]):
    customers_copy[col] = customers_copy[col].astype(np.int64) // 10**9

# Assuming that these are the categorical columns in your dataset
categorical_cols = ['job', 'marital_status', 'education_level', 'housing_loan_status', 'personal_loan_status']

# Transform categorical variables to dummy variables
transformer = make_column_transformer(
    (OneHotEncoder(), categorical_cols),
    remainder='passthrough'
)

X = transformer.fit_transform(customers_copy.drop('default_history', axis=1))
y = customers_copy['default_history']

# Use SMOTE to balance the classes
smote = SMOTE(sampling_strategy='minority')
X_sm, y_sm = smote.fit_resample(X, y)

# Check the balance of the classes after the oversampling
print(pd.Series(y_sm).value_counts())

import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LogisticRegression
from sklearn.preprocessing import StandardScaler, OneHotEncoder
from sklearn.compose import make_column_transformer
from sklearn.pipeline import make_pipeline
from sklearn.impute import SimpleImputer

# Ensure 'date' column in transactions is in datetime format
transactions['date'] = pd.to_datetime(transactions['date'])

# Create 'month_year' feature for aggregating transactions on a monthly basis
transactions['month_year'] = transactions['date'].dt.to_period('M')

# Compute Transaction Frequency and Average Transaction Amount
transaction_frequency = transactions.groupby(['customer_id', 'month_year'])['transaction_id'].count().reset_index().rename(columns={"transaction_id": "transaction_frequency"})
average_transaction_amount = transactions.groupby(['customer_id', 'month_year'])['transaction_amount'].mean().reset_index().rename(columns={"transaction_amount": "average_transaction_amount"})

# Compute Monthly Balance Trend as the difference between the maximum and minimum balance in the month
balance_max = transactions.groupby(['customer_id', 'month_year'])['balance_after_transaction'].max().reset_index().rename(columns={"balance_after_transaction": "balance_max"})
balance_min = transactions.groupby(['customer_id', 'month_year'])['balance_after_transaction'].min().reset_index().rename(columns={"balance_after_transaction": "balance_min"})
monthly_balance_trend = pd.merge(balance_max, balance_min, on=['customer_id', 'month_year'])
monthly_balance_trend['monthly_balance_trend'] = monthly_balance_trend['balance_max'] - monthly_balance_trend['balance_min']

# Merge all dynamic features
dynamic_features = pd.merge(pd.merge(transaction_frequency, average_transaction_amount, on=['customer_id', 'month_year']), monthly_balance_trend, on=['customer_id', 'month_year'])

# Merge dynamic features with customer data
df = pd.merge(customers, dynamic_features, on='customer_id', how='left')

# Assuming 'default_history' is the target variable
# Convert 'default_history' into binary format
df['default_history'] = df['default_history'].map({'No': 0, 'Yes': 1})

# Define preprocessor
preprocessor = make_column_transformer(
    (make_pipeline(SimpleImputer(strategy='mean'), StandardScaler()),
     ['age', 'credit_score', 'transaction_frequency', 'average_transaction_amount', 'monthly_balance_trend']),
    (OneHotEncoder(drop='first'), ['job', 'marital_status', 'education_level', 'housing_loan_status', 'personal_loan_status'])
)

# Define model
model = make_pipeline(
    preprocessor,
    LogisticRegression(solver='liblinear')
)

# Split data into training and test sets
X_train, X_test, y_train, y_test = train_test_split(
    df.drop(['customer_id', 'default_history', 'month_year', 'balance_max', 'balance_min'], axis=1),
    df['default_history'],
    test_size=0.2,
    random_state=0
)

# Train model
model.fit(X_train, y_train)

# Make predictions on test set
y_pred = model.predict(X_test)

from sklearn.ensemble import RandomForestClassifier, VotingClassifier

# Define preprocessor for the logistic regression
preprocessor1 = make_column_transformer(
    (make_pipeline(SimpleImputer(strategy='mean'), StandardScaler()),
     ['age', 'credit_score', 'transaction_frequency', 'average_transaction_amount', 'monthly_balance_trend']),
    (OneHotEncoder(drop='first'), ['job', 'marital_status', 'education_level', 'housing_loan_status', 'personal_loan_status'])
)

# Define preprocessor for the random forest
preprocessor2 = make_column_transformer(
    (make_pipeline(SimpleImputer(strategy='mean'), StandardScaler()),
     ['age', 'credit_score']),
    (OneHotEncoder(drop='first'), ['job', 'marital_status', 'education_level', 'housing_loan_status', 'personal_loan_status'])
)

# Define the logistic regression model
model1 = make_pipeline(
    preprocessor1,
    LogisticRegression(solver='liblinear')
)

# Define the random forest model
model2 = make_pipeline(
    preprocessor2,
    RandomForestClassifier(n_estimators=100, random_state=0)
)

# Define the voting classifier, with two estimators: logistic regression and random forest
voting_clf = VotingClassifier(
    estimators=[('lr', model1), ('rf', model2)],
    voting='soft'
