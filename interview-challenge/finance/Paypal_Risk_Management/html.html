import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# Load data from CSV
df_UserRiskProfile = pd.read_csv("https://storage.googleapis.com/datalynn-datasets/interview-challenge/technology/Paypal_Risk_Management/Riskdata.csv")

# Custom function to convert string to list
def str_to_list(s):
    if isinstance(s, str):
        return list(map(int, s[1:-1].split()))
    return s

# Convert string representation to list
df_UserRiskProfile['TransactionHist'] = df_UserRiskProfile['TransactionHist'].apply(str_to_list)

# Handle cases where TransactionHist is empty or None
def safe_mean(lst):
    return np.mean(lst) if lst else 0

def safe_len(lst):
    return len(lst) if lst else 0

# Generate a new feature: Average Transaction Amount for each user
df_UserRiskProfile['MeanTransactionAmount'] = df_UserRiskProfile['TransactionHist'].apply(safe_mean)

# Generate a new feature: Transaction Frequency for each user
df_UserRiskProfile['TransactionFrequency'] = df_UserRiskProfile['TransactionHist'].apply(safe_len)

# Plot Mean Transaction Amount vs. Chargeback History
plt.figure()
plt.title("Mean Transaction Amount vs. Chargeback History")
plt.scatter(df_UserRiskProfile['MeanTransactionAmount'], df_UserRiskProfile['ChargebackHist'], alpha=0.5)
plt.xlabel('Mean Transaction Amount')
plt.ylabel('Number of Chargebacks')
plt.show()

# Plot Transaction Frequency vs. Chargeback History
plt.figure()
plt.title("Transaction Frequency vs. Chargeback History")
plt.scatter(df_UserRiskProfile['TransactionFrequency'], df_UserRiskProfile['ChargebackHist'], alpha=0.5)
plt.xlabel('Transaction Frequency')
plt.ylabel('Number of Chargebacks')
plt.show()

# Compute the correlation matrix
correlation_matrix = df_UserRiskProfile[['MeanTransactionAmount', 'TransactionFrequency', 'ChargebackHist']].corr()

from sklearn.ensemble import RandomForestClassifier
from sklearn.model_selection import train_test_split
from sklearn.metrics import classification_report
import numpy as np

# Feature engineering for TransactionHist
# For simplicity, let's calculate the mean and standard deviation of past transactions for each user
df_UserRiskProfile['MeanTransactionAmount'] = df_UserRiskProfile['TransactionHist'].apply(np.mean)
df_UserRiskProfile['StdTransactionAmount'] = df_UserRiskProfile['TransactionHist'].apply(np.std)

# Prepare features and target variable
X = df_UserRiskProfile[['CreditScore', 'MeanTransactionAmount', 'StdTransactionAmount', 'LoanOffered', 'FeedbackScore', 'ChargebackHist', 'MerchantRating', 'ThirdPartyData']]
y = df_UserRiskProfile['LoanDefaulted']

# Train-test split
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)

# Model training
clf = RandomForestClassifier(random_state=42)
clf.fit(X_train, y_train)

# Prediction and evaluation
y_pred = clf.predict(X_test)
print("Classification Report:\n", classification_report(y_test, y_pred))

# Feature Importances
print("Feature importances:", clf.feature_importances_)

For instance, users with a certain CreditScore range might benefit from a different loan offering.

Solution:¶

Hints:¶

  • Feature Selection: Carefully choose the variables that capture various aspects of the user's financial behavior and risk profile. It's essential to pick features that are likely to distinguish different user clusters effectively.
  • Data Preprocessing: Ensure that the data is normalized or standardized, as K-means clustering is sensitive to the scale of data.
  • Number of Clusters: The choice of the number of clusters (n_clusters in KMeans) can significantly impact the segmentation. You can use techniques like the Elbow Method to determine the optimal number of clusters.
  • Product Matching: After identifying the clusters, think from a business perspective to match the identified user needs with suitable products. It's not just about the technical accuracy of clusters but also about how actionable they are.

from sklearn.cluster import KMeans
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import silhouette_score

# Calculate the mean transaction amount for each user and add it as a feature
df_UserRiskProfile['MeanTransactionAmount'] = df_UserRiskProfile['TransactionHist'].apply(lambda x: np.mean(x))

# Drop non-numeric and target variables for clustering
X = df_UserRiskProfile.drop(['UserID', 'TransactionHist', 'TransactionDate', 'LoanDefaulted'], axis=1)

# Normalize features
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

# Initialize and fit KMeans
best_n_clusters = 3  # Use silhouette score or elbow method to find the best number of clusters
kmeans = KMeans(n_clusters=best_n_clusters, n_init=10)
kmeans.fit(X_scaled)

# Add cluster labels to the original dataframe
df_UserRiskProfile['Cluster'] = kmeans.labels_

# Display mean values for each cluster
# Explicitly specify numeric_only to suppress warning
cluster_summary = df_UserRiskProfile.groupby('Cluster').mean(numeric_only=True)
print(cluster_summary)

# Calculate silhouette score
print("Silhouette Score:", silhouette_score(X_scaled, kmeans.labels_))
